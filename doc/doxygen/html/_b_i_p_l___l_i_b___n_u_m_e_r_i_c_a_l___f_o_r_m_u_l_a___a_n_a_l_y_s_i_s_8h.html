MultiplicationAndDivision :: = <Pulus_MinusSign> | {<Pulus_MinusSign> {*|/} <MultiplicationAndDivision>}

Power :: = <Factor> | {<Power> {^} <Factor>}
Factor :: = <Number> | (<AdditionAndSubtraction>)
