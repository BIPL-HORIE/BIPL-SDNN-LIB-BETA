乱数の使い方の例

unsigned int three[3] = { 2146876,324987,23196879 }; BIPL_RANDOM_UNIFORM_DOUBLE testrand; testrand.Init_BIPL_RANDOM(0, 1, three, 3); for (int i = 0; i < 10; i++) std::cout << testrand << std::endl;

BIPL_RANDOM_UNIFORM_INT testrand2; testrand2.Init_BIPL_RANDOM(0, 10, three, 3); for (int i = 0; i < 10; i++) std::cout << testrand2 << std::endl;
